🅰 frontend/src/app/interceptors/auth.interceptor.ts — Automatically attaches JWT to every HTTP request 
export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const auth  = inject(AuthService);
  const token = auth.getToken();

  if (token) {
    req = req.clone({
      setHeaders: { Authorization: `Bearer ${token}` }
    });
  }
  return next(req);
};

PostGIS PointField · nearby_drivers query · TripConsumer WebSocket · Google Maps integration

🗺 PostGIS Spatial Query

  • PointField(geography=True) for accurate distance
  • distance_lte filter with D(km=radius)
  • annotate(distance=Distance(...)) for ordering
  • Returns 20 nearest available drivers
  • SRID 4326 (WGS84 — standard GPS)

⚡ WebSocket Channels

  • ws/trip/<id>/ — per-trip room (rider + driver)
  • ws/map/ — broadcast driver positions to riders
  • Redis channel layer for multi-process pub/sub
  • Event types: location_update, status_update, chat
  • AsyncWebsocketConsumer (non-blocking)
📍 backend/drivers/views.py — nearby_drivers() — PostGIS spatial query for finding available drivers 
@api_view(["GET"])
@permission_classes([IsAuthenticated])
def nearby_drivers(request):
    lat    = float(request.query_params["lat"])
    lng    = float(request.query_params["lng"])
    radius = float(request.query_params.get("radius", 5))

    user_location = Point(lng, lat, srid=4326)

    drivers = (
        DriverProfile.objects
        .filter(status=DriverProfile.Status.AVAILABLE, location__isnull=False)
        .filter(location__distance_lte=(user_location, D(km=radius)))
        .annotate(distance=Distance("location", user_location))
        .order_by("distance")[:20]
    )
    return Response(DriverProfileSerializer(drivers, many=True).data)

🔌 backend/trips/consumers.py — TripConsumer — Django Channels WebSocket consumer for real-time trip events 
class TripConsumer(AsyncWebsocketConsumer):
    async def connect(self):
        self.group_name = f"trip_{self.scope['url_route']['kwargs']['trip_id']}"
        await self.channel_layer.group_add(self.group_name, self.channel_name)
        await self.accept()

    async def receive(self, text_data=None, ...):
        data = json.loads(text_data)
        if data["type"] == "location_update":
            # Broadcast driver GPS to all group members
            await self.channel_layer.group_send(self.group_name, {
                "type":     "driver_location",
                "latitude": data["latitude"],
                "longitude":data["longitude"],
            })

🗺 frontend/src/app/services/websocket.service.ts — Angular WebSocket service wrapping native browser WebSocket API 
@Injectable({ providedIn: 'root' })
export class WebSocketService {
  private messages$ = new Subject<WsMessage>();

  connectToTrip(tripId: number): void {
    this.connect(`ws://localhost:8001/ws/trip/${tripId}/`);
  }

  sendLocation(latitude: number, longitude: number): void {
    this.send({ type: 'location_update', latitude, longitude });
  }

  getMessages(): Observable<WsMessage> {
    return this.messages$.asObservable();
  }
}

Trip Lifecycle, Stripe Payments & Angular Booking UI

Request/Accept/Complete/Cancel · Haversine fare estimation · PaymentIntent · Booking components

🚕 Trip Status Machine

  • REQUESTED → fare estimated via Haversine
  • ACCEPTED → driver assigned, driver goes ON_TRIP
  • IN_TRANSIT → en route to destination
  • COMPLETED → driver freed, total_trips ++
  • CANCELLED → available at any pre-complete stage

💳 Stripe Payment Flow

  • POST /api/payments/create-intent/ → client_secret
  • Angular uses client_secret with Stripe.js
  • Webhook updates Payment.status on confirmation
  • Amount in KZT (tiyn = smallest unit same as KZT)
  • Webhook signature verified in production
🚗 backend/trips/views.py — request_trip() — Haversine distance + fare calculation on trip request 
def haversine_km(lat1, lng1, lat2, lng2):
    R = 6371
    phi1, phi2 = math.radians(lat1), math.radians(lat2)
    dphi = math.radians(lat2 - lat1)
    dlam = math.radians(lng2 - lng1)
    a = math.sin(dphi/2)**2 + math.cos(phi1)*math.cos(phi2)*math.sin(dlam/2)**2
    return 2 * R * math.asin(math.sqrt(a))

def calculate_fare(distance_km):
    return round(300 + distance_km * 120, 2)  # 300₸ base + 120₸/km

💳 backend/payments/views.py — create_payment_intent() — Stripe PaymentIntent creation for frontend payment confirmation 
@api_view(["POST"])
def create_payment_intent(request):
    trip = Trip.objects.get(pk=trip_id, rider=request.user)
    amount_tiyn = int(trip.fare_estimate * 100)

    intent = stripe.PaymentIntent.create(
        amount      = amount_tiyn,
        currency    = "kzt",
        description = f"TaxiApp Trip #{trip.pk}",
        metadata    = {"trip_id": trip.pk},
    )
    return Response({"client_secret": intent["client_secret"], "amount": trip.fare_estimate})
